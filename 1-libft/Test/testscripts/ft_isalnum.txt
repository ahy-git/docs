/*
	Resumo dos testes:
	1. Testa letras maiusculas, minusculas e digitos validos.
	2. Confirma que o retorno da ft_isalnum deve ser exatamente 1.
	3. Testa caracteres que nao sao alfanumericos.
	4. Testa limites dos intervalos: 'Z', 'a', '0' e '9'.
	5. Compara com isalnum original convertendo retorno para 0 ou 1.
*/

#include "libft.h"
#include <ctype.h>
#include <stdio.h>

int	main(void)
{
	printf("Test: isalnum('A' 'z' '5') | letras e digito validos\n");
	printf("      subject: retorno deve ser 1, nao apenas nao-zero\n");
	printf("      entrada : 'A'=65, 'z'=122, '5'=53\n");
	printf("      esperado: 1 1 1\n");
	printf("FT  : %d %d %d\n",
		ft_isalnum('A'), ft_isalnum('z'), ft_isalnum('5'));
	printf("ORIG: %d %d %d\n",
		isalnum('A') != 0, isalnum('z') != 0, isalnum('5') != 0);
	printf("\n");
	printf("Test: isalnum('!' ' ' 0 127) | nao-alfanumericos\n");
	printf("      caracteres especiais, nulo e DEL devem retornar 0\n");
	printf("      entrada : '!'=33, ' '=32, 0, 127\n");
	printf("      esperado: 0 0 0 0\n");
	printf("FT  : %d %d %d %d\n",
		ft_isalnum('!'), ft_isalnum(' '),
		ft_isalnum(0), ft_isalnum(127));
	printf("ORIG: %d %d %d %d\n",
		isalnum('!') != 0, isalnum(' ') != 0,
		isalnum(0) != 0, isalnum(127) != 0);
	printf("\n");
	printf("Test: isalnum('Z' 'a' '0' '9') | limites dos intervalos\n");
	printf("      ultimo maiusculo, primeiro minusculo, digitos extremos\n");
	printf("      entrada : 'Z'=90, 'a'=97, '0'=48, '9'=57\n");
	printf("      esperado: 1 1 1 1\n");
	printf("FT  : %d %d %d %d\n",
		ft_isalnum('Z'), ft_isalnum('a'),
		ft_isalnum('0'), ft_isalnum('9'));
	printf("ORIG: %d %d %d %d\n",
		isalnum('Z') != 0, isalnum('a') != 0,
		isalnum('0') != 0, isalnum('9') != 0);
	return (0);
}
