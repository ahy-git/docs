/*
	Resumo dos testes:
	1. Testa os limites validos do intervalo ASCII: 0, 65 e 127.
	2. Testa valores fora do intervalo ASCII: 128 e 255.
	3. Testa valor negativo, que deve retornar 0.
	4. Testa valores internos proximos dos limites: 1 e 126.
	5. Compara com isascii original da libc.
*/

#include "libft.h"
#include <ctype.h>
#include <stdio.h>

int	main(void)
{
	printf("Test: isascii(0 65 127) | limites do intervalo ASCII\n");
	printf("      0=nulo, 65='A', 127=DEL sao todos ASCII validos\n");
	printf("      entrada : 0, 65, 127\n");
	printf("      esperado: 1 1 1\n");
	printf("FT  : %d %d %d\n",
		ft_isascii(0), ft_isascii(65), ft_isascii(127));
	printf("ORIG: %d %d %d\n",
		isascii(0), isascii(65), isascii(127));
	printf("\n");
	printf("Test: isascii(128 255) | primeiro e ultimo nao-ASCII\n");
	printf("      128 e 255 estao fora do intervalo 0-127\n");
	printf("      entrada : 128, 255\n");
	printf("      esperado: 0 0\n");
	printf("FT  : %d %d\n",
		ft_isascii(128), ft_isascii(255));
	printf("ORIG: %d %d\n",
		isascii(128), isascii(255));
	printf("\n");
	printf("Test: isascii(-1) | valor negativo\n");
	printf("      char negativo: armadilha classica da Moulinette\n");
	printf("      entrada : -1\n");
	printf("      esperado: 0\n");
	printf("FT  : %d\n", ft_isascii(-1));
	printf("ORIG: %d\n", isascii(-1));
	printf("\n");
	printf("Test: isascii(1 126) | vizinhos internos dos limites\n");
	printf("      logo apos o nulo e logo antes do DEL\n");
	printf("      entrada : 1, 126='~'\n");
	printf("      esperado: 1 1\n");
	printf("FT  : %d %d\n",
		ft_isascii(1), ft_isascii(126));
	printf("ORIG: %d %d\n",
		isascii(1), isascii(126));
	return (0);
}
