/*
	Resumo dos testes:
	1. Testa digitos validos: '0', '5' e '9'.
	2. Confirma que a ft_isdigit deve retornar exatamente 1.
	3. Testa caracteres que nao sao digitos: letras, espaco e nulo.
	4. Testa os vizinhos dos limites: '/' antes de '0' e ':' apos '9'.
	5. Testa valores fora do ASCII: -1, 128 e 255.
	6. Compara com isdigit original convertendo retorno para 0 ou 1.
*/

#include "libft.h"
#include <ctype.h>
#include <stdio.h>

int	main(void)
{
	printf("Test: isdigit('0' '5' '9') | digitos validos\n");
	printf("      subject: retorno deve ser 1, nao apenas nao-zero\n");
	printf("      entrada : '0'=48, '5'=53, '9'=57\n");
	printf("      esperado: 1 1 1\n");
	printf("FT  : %d %d %d\n",
		ft_isdigit('0'), ft_isdigit('5'), ft_isdigit('9'));
	printf("ORIG: %d %d %d\n",
		isdigit('0') != 0, isdigit('5') != 0, isdigit('9') != 0);
	printf("\n");
	printf("Test: isdigit('a' 'Z' ' ' 0) | nao-digitos\n");
	printf("      letra, maiuscula, espaco e nulo devem retornar 0\n");
	printf("      entrada : 'a'=97, 'Z'=90, ' '=32, 0\n");
	printf("      esperado: 0 0 0 0\n");
	printf("FT  : %d %d %d %d\n",
		ft_isdigit('a'), ft_isdigit('Z'),
		ft_isdigit(' '), ft_isdigit(0));
	printf("ORIG: %d %d %d %d\n",
		isdigit('a') != 0, isdigit('Z') != 0,
		isdigit(' ') != 0, isdigit(0) != 0);
	printf("\n");
	printf("Test: isdigit(47 58) | vizinhos dos limites '0' e '9'\n");
	printf("      47='/' esta antes de '0'=48, 58=':' apos '9'=57\n");
	printf("      entrada : 47, 58\n");
	printf("      esperado: 0 0\n");
	printf("FT  : %d %d\n",
		ft_isdigit(47), ft_isdigit(58));
	printf("ORIG: %d %d\n",
		isdigit(47) != 0, isdigit(58) != 0);
	printf("\n");
	printf("Test: isdigit(-1 128 255) | valores fora do ASCII\n");
	printf("      negativos e acima de 127 devem retornar 0\n");
	printf("      entrada : -1, 128, 255\n");
	printf("      esperado: 0 0 0\n");
	printf("FT  : %d %d %d\n",
		ft_isdigit(-1), ft_isdigit(128), ft_isdigit(255));
	printf("ORIG: %d %d %d\n",
		isdigit(-1) != 0, isdigit(128) != 0, isdigit(255) != 0);
	return (0);
}
