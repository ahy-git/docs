/*
	Resumo dos testes:
	1. Testa ft_lstadd_back em lista vazia adicionando A, B e C.
	2. Verifica se os nos entram no final preservando a ordem.
	3. Testa mistura de ft_lstadd_back com ft_lstadd_front.
	4. Verifica se a lista final continua na ordem esperada: A B C.
	5. Testa protecao com lst=NULL para nao crashar.
*/

#include "libft.h"
#include <stdio.h>
#include <stdlib.h>

static void	free_list(t_list *lst)
{
	t_list	*tmp;

	while (lst)
	{
		tmp = lst->next;
		free(lst);
		lst = tmp;
	}
}

static void	print_contents(char *label, t_list *lst)
{
	printf("%s", label);
	while (lst)
	{
		printf(" %s", (char *)lst->content);
		lst = lst->next;
	}
	printf("\n");
}

static void	test_insercao(void)
{
	t_list	*lst;

	lst = NULL;
	ft_lstadd_back(&lst, ft_lstnew("A"));
	ft_lstadd_back(&lst, ft_lstnew("B"));
	ft_lstadd_back(&lst, ft_lstnew("C"));
	printf("Test: ft_lstadd_back x3 em lista vazia\n");
	printf("      cada no deve ir para o final, preservando ordem\n");
	printf("      entrada : NULL -> back(A) -> back(B) -> back(C)\n");
	printf("      esperado: A B C\n");
	print_contents("FT  :", lst);
	printf("ORIG: A B C\n");
	free_list(lst);
	printf("\n");
}

static void	test_mix_e_null(void)
{
	t_list	*lst;
	t_list	*tmp_node;

	lst = NULL;
	ft_lstadd_back(&lst, ft_lstnew("B"));
	ft_lstadd_front(&lst, ft_lstnew("A"));
	ft_lstadd_back(&lst, ft_lstnew("C"));
	printf("Test: back(B) front(A) back(C) | mix de insercoes\n");
	printf("      add_front insere no inicio, add_back no final\n");
	printf("      entrada : back(B), front(A), back(C)\n");
	printf("      esperado: A B C\n");
	print_contents("FT  :", lst);
	printf("ORIG: A B C\n");
	free_list(lst);
	printf("\n");
	tmp_node = ft_lstnew("X");
	ft_lstadd_back(NULL, tmp_node);
	printf("Test: ft_lstadd_back(NULL, node) | lst=NULL\n");
	printf("      deve proteger contra lst nulo sem crashar\n");
	printf("      entrada : lst=NULL, node valido\n");
	printf("      esperado: sem crash\n");
	printf("FT  : sem crash\n");
	printf("ORIG: sem crash\n");
	free(tmp_node);
}

int	main(void)
{
	test_insercao();
	test_mix_e_null();
	return (0);
}
