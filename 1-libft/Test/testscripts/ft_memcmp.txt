/*
	Resumo dos testes:
	1. Testa blocos iguais, que devem retornar 0.
	2. Testa diferenca depois do limite n, que deve ser ignorada.
	3. Testa n=0, que sempre deve retornar 0.
	4. Testa caso em que s1 e menor que s2.
	5. Testa caso em que s1 e maior que s2.
	6. Testa comparacao depois do byte nulo.
	7. Compara com memcmp original da libc usando sinal normalizado.
*/

#include "libft.h"
#include <stdio.h>
#include <string.h>

static int	normalize(int n)
{
	if (n < 0)
		return (-1);
	if (n > 0)
		return (1);
	return (0);
}

static void	test_igualdade(void)
{
	printf("Test: ft_memcmp(\"abc\",\"abc\",3) | strings iguais\n");
	printf("      comparacao exata de 3 bytes identicos\n");
	printf("      entrada : s1=\"abc\", s2=\"abc\", n=3\n");
	printf("      esperado: 0\n");
	printf("FT  : %d\n", ft_memcmp("abc", "abc", 3));
	printf("ORIG: %d\n", memcmp("abc", "abc", 3));
	printf("\n");
	printf("Test: ft_memcmp(\"abc\",\"abd\",2) | diferem apos n\n");
	printf("      so 2 bytes comparados, diferenca em s[2] ignorada\n");
	printf("      entrada : s1=\"abc\", s2=\"abd\", n=2\n");
	printf("      esperado: 0\n");
	printf("FT  : %d\n", ft_memcmp("abc", "abd", 2));
	printf("ORIG: %d\n", memcmp("abc", "abd", 2));
	printf("\n");
	printf("Test: ft_memcmp(s, s, 0) | n=0\n");
	printf("      zero bytes comparados, sempre retorna 0\n");
	printf("      entrada : s1=\"abc\", s2=\"xyz\", n=0\n");
	printf("      esperado: 0\n");
	printf("FT  : %d\n", ft_memcmp("abc", "xyz", 0));
	printf("ORIG: %d\n", memcmp("abc", "xyz", 0));
	printf("\n");
}

static void	test_diferenca(void)
{
	int	ft_r;
	int	orig_r;

	printf("Test: ft_memcmp(\"abc\",\"abd\",3) | s1 menor que s2\n");
	printf("      'c'=99 < 'd'=100, resultado deve ser negativo\n");
	printf("      entrada : s1=\"abc\", s2=\"abd\", n=3\n");
	printf("      esperado: neg (-1 normalizado)\n");
	ft_r = normalize(ft_memcmp("abc", "abd", 3));
	orig_r = normalize(memcmp("abc", "abd", 3));
	printf("FT  : %d\n", ft_r);
	printf("ORIG: %d\n", orig_r);
	printf("\n");
	printf("Test: ft_memcmp(\"abd\",\"abc\",3) | s1 maior que s2\n");
	printf("      'd'=100 > 'c'=99, resultado deve ser positivo\n");
	printf("      entrada : s1=\"abd\", s2=\"abc\", n=3\n");
	printf("      esperado: pos (1 normalizado)\n");
	ft_r = normalize(ft_memcmp("abd", "abc", 3));
	orig_r = normalize(memcmp("abd", "abc", 3));
	printf("FT  : %d\n", ft_r);
	printf("ORIG: %d\n", orig_r);
	printf("\n");
}

static void	test_apos_nulo(void)
{
	int	ft_r;
	int	orig_r;

	printf("Test: ft_memcmp(\"abc\\0def\",\"abc\\0xyz\",7) | alem nulo\n");
	printf("      memcmp ve bytes puros, diferenca apos \\0 conta\n");
	printf("      entrada : s1=\"abc\\0def\", s2=\"abc\\0xyz\", n=7\n");
	printf("      esperado: neg ('d'=100 < 'x'=120)\n");
	ft_r = normalize(ft_memcmp("abc\0def", "abc\0xyz", 7));
	orig_r = normalize(memcmp("abc\0def", "abc\0xyz", 7));
	printf("FT  : %d\n", ft_r);
	printf("ORIG: %d\n", orig_r);
}

int	main(void)
{
	test_igualdade();
	test_diferenca();
	test_apos_nulo();
	return (0);
}
