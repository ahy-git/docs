
//A função ft_putstr_fd escreve uma string inteira 
//no file descriptor usando write. Em vez de fazer 
//um loop chamando ft_putchar_fd para cada caractere, 
//faço uma única chamada a write passando o tamanho calculado 
//por ft_strlen — uma syscall em vez de várias, mais eficiente. 
//O s já é char *, então não precisa de cast. O retorno de 
//write é ignorado conforme o protótipo void do enunciado."

#include "libft.h"
#include <stdio.h>
#include <unistd.h>

static void	print_result(char *label, char *buf, ssize_t count)
{
	int	i;

	printf("%s count=%zd bytes=", label, count);
	i = 0;
	while (i < count)
	{
		printf("%u", (unsigned char)buf[i]);
		if (i + 1 < count)
			printf(" ");
		i++;
	}
	printf("\n");
}

static void	test_putstr(char *s, char *name, char *expected)
{
	int		fd[2];
	char	buf[100];
	ssize_t	count;

	printf("Test: ft_putstr_fd(%s, fd) | escreve string\n", name);
	printf("      escreve a string sem adicionar newline\n");
	printf("      entrada : s=%s\n", name);
	printf("      esperado: %s\n", expected);
	if (pipe(fd) == -1)
	{
		printf("FT  : pipe error\n");
		printf("ORIG: %s\n", expected);
		return ;
	}
	ft_putstr_fd(s, fd[1]);
	close(fd[1]);
	count = read(fd[0], buf, 100);
	close(fd[0]);
	print_result("FT  :", buf, count);
	printf("ORIG: %s\n", expected);
	printf("\n");
}

int	main(void)
{
	test_putstr("Hello", "\"Hello\"",
		"count=5 bytes=72 101 108 108 111");
	test_putstr("", "\"\"",
		"count=0 bytes=");
	test_putstr("a\nb", "\"a\\nb\"",
		"count=3 bytes=97 10 98");
	test_putstr("ERR", "\"ERR\"",
		"count=3 bytes=69 82 82");
	return (0);
}
